288 void tune_pwm_freq() {
289  // Setup interrupt for frequency measurement
290  const uint8_t adj_pwm_pin = 10;
291  EICRA |= bit(ISC10) | bit(ISC11); // Set rising edge interrupts on pin D3
292  EIMSK |= bit (INT1); // enable interrupt 1 pin D3
293 
294  // Start pwm and measure frequency
295  external_pwm.setPWM(adj_pwm_pin, 0, 2000); // Start pwm signal on pin 10
296  const float smoothing = 5.0; // number of measurements to use
297  // Measure frequency
298  float interval_sum = 0;
299  for (int i = 0; i < smoothing; i++){
300  delay(30);
301  interval_sum += PWM_INTERVAL_MEASURED;
302  }
303  // Set frequency as close to the desired as possible
304  if (interval_sum > 1000 && interval_sum < 1e6){ // detect pin 10 connection
305  float freq_observed = (1000000.0*smoothing)/(interval_sum);
306  float alpha = PWM_FREQUENCY/freq_observed;
307  external_pwm.setPWMFreq(alpha*PWM_FREQUENCY);
308  // Measure frequency again and adjust PWM ticks accordingly
309  // needed since the frequency prescaler only has 8 bit resolution
310  interval_sum = 0;
311  delay(30);
312  for (int i = 0; i < smoothing; i++){
313  delay(30);
314  interval_sum += PWM_INTERVAL_MEASURED;
315  }
316  freq_observed = (1000000.0*smoothing)/(interval_sum);
317  unsigned long pwm_min_tick = DUTY_CYCLE_MIN*freq_observed*PWM_RES;
318  unsigned long pwm_max_tick = DUTY_CYCLE_MAX*freq_observed*PWM_RES;
319  PWM_NEUTRAL_TICK = int((pwm_min_tick + (pwm_max_tick - pwm_min_tick)*0.5)/1000.0 + 0.5);
320  INPUT_SCALE = ((freq_observed*PWM_RES*DUTY_CYCLE_MIN)/1000.0 - PWM_NEUTRAL_TICK)/(float)INPUT_MIN;
321  }
322  else {;}
323  EIMSK &= ~bit (INT1); // Disable interrupt on D3
324 }
